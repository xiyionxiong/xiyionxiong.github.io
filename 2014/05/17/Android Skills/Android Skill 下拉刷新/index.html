    /**
     * еңЁиў«еҲӨе®ҡдёәж»ҡеҠЁд№ӢеүҚз”ЁжҲ·жүӢжҢҮеҸҜд»Ҙз§»еҠЁзҡ„жңҖеӨ§еҖј

     * жҳҜеҗҰе·ІеҠ иҪҪиҝҮдёҖж¬ЎlayoutпјҢиҝҷйҮҢonLayoutдёӯзҡ„еҲқе§ӢеҢ–еҸӘйңҖеҠ иҪҪдёҖж¬Ў

     * еҪ“еүҚжҳҜеҗҰеҸҜд»ҘдёӢжӢүпјҢеҸӘжңүListViewж»ҡеҠЁеҲ°еӨҙзҡ„ж—¶еҖҷжүҚе…Ғи®ёдёӢжӢү
     */
    private boolean ableToPull;

    /**
     * head View еҢ…еҗ«зҡ„дёҖдёӘ LinearLayout
     */
    private LinearLayout pull_to_refresh_linearLayout;

    /**
     * дёӢжӢүеҲ·ж–°жҺ§д»¶зҡ„жһ„йҖ еҮҪж•°пјҢдјҡеңЁиҝҗиЎҢж—¶еҠЁжҖҒж·»еҠ дёҖдёӘдёӢжӢүеӨҙзҡ„еёғеұҖгҖӮ
     *
     * @param context
     * @param attrs
     */
    @SuppressLint("InflateParams")
    public PullToRefreshView(Context context, AttributeSet attrs) {

        super(context, attrs);

        // е®һдҫӢеҢ–SharedPreferences
        this.preferences = PreferenceManager
                .getDefaultSharedPreferences(context);

        // е®һдҫӢеҢ– дёӢжӢүеӨҙзҡ„ еёғеұҖView
        this.header = LayoutInflater.from(context).inflate(
                R.layout.pull_to_refresh, null, true);

        // е®һдҫӢеҢ– дёӢжӢүеӨҙзҡ„ LinearLayout
        this.pull_to_refresh_linearLayout = (LinearLayout) this.header
                .findViewById(R.id.pull_to_refresh_linearLayout);

        // еҸ–еҫ— дёӢжӢүеӨҙзҡ„ еёғеұҖ йҮҢзҡ„ з»„д»¶
        this.progressBar = (ProgressBar) this.header
                .findViewById(R.id.progress_bar);

        this.arrow = (ImageView) this.header.findViewById(R.id.arrow);

        this.description = (TextView) this.header
                .findViewById(R.id.description);

        this.updateAt = (TextView) this.header.findViewById(R.id.updated_at);

        // еҸ–еҫ—з§»еҠЁзҡ„жңҖе°Ҹи·қзҰ»,еҸӘжңүеӨ§дәҺиҝҷдёӘи·қзҰ»,жүҚеҸҜи®ӨдёәжҳҜ move
        this.touchSlop = ViewConfiguration.get(context).getScaledTouchSlop();

        // и®ҫзҪ®дёҠж¬Ўжӣҙж–°ж–Үеӯ—жҸҸиҝ°s
        this.refreshUpdatedAtValue();

        // и®ҫзҪ®еёғеұҖдёәеһӮзӣҙ
        setOrientation(VERTICAL);

        // дёӢжӢүеӨҙзҡ„View зҪ®йЎ¶
        addView(this.header, 0);

    }

    /**
     * иҝӣиЎҢдёҖдәӣе…ій”®жҖ§зҡ„еҲқе§ӢеҢ–ж“ҚдҪң
     *
     * жҜ”еҰӮпјҡе°ҶдёӢжӢүеӨҙеҗ‘дёҠеҒҸз§»иҝӣиЎҢйҡҗи—ҸпјҢз»ҷListViewжіЁеҶҢtouchдәӢд»¶гҖӮ
     */
    @Override
    protected void onLayout(boolean changed, int l, int t, int r, int b) {

        super.onLayout(changed, l, t, r, b);

        // еҰӮжһңloadOnceжІЎжңүи®ҫзҪ®(жІЎжңүеҠ иҪҪиҝҮдёҖж¬Ўlayout)
        if (changed && !this.loadOnce) {

            this.hideHeaderHeight = -this.header.getHeight();

            this.headerLayoutParams = (MarginLayoutParams) this.header
                    .getLayoutParams();

            this.headerLayoutParams.topMargin = this.hideHeaderHeight;

            // getChildAt(1); жҳҜ 1 дёҚжҳҜ L
            this.listView = (ListView) getChildAt(1);

            System.out.println("this.listView = " + this.listView);

            // и®ҫзҪ®ListViewзҡ„и§Ұж‘ёдәӢд»¶
            this.listView.setOnTouchListener(this);

            // жӣҙеҸҳ loadOnce зҡ„еҖј
            this.loadOnce = true;

        }

    }

    public static boolean result = false;

    public static boolean state = false;

    /**
     * еҪ“ListViewиў«и§Ұж‘ёж—¶и°ғз”ЁпјҢе…¶дёӯеӨ„зҗҶдәҶеҗ„з§ҚдёӢжӢүеҲ·ж–°зҡ„е…·дҪ“йҖ»иҫ‘гҖӮ
     */
    @SuppressLint("ClickableViewAccessibility")
    @Override
    public boolean onTouch(View v, MotionEvent event) {

        /**
         * и®ҫе®ҡ {@link #ableToPull}
         */
        this.setIsAbleToPull(event);

        // еҰӮжһңе…Ғи®ёдёӢжӢү
        if (this.ableToPull) {

            switch (event.getAction()) {

            // "жҢүдёӢ"
            case MotionEvent.ACTION_DOWN:

                PullToRefreshView.result = false;

                PullToRefreshView.state = false;

                this.yDown = event.getRawY();

                break;

            case MotionEvent.ACTION_MOVE:

                if (!PullToRefreshView.state) {

                    PullToRefreshView.result = true;

                    float yMove = event.getRawY();

                    int distance = (int) (yMove - this.yDown);

                    /**
                     * еҰӮжһңжүӢжҢҮжҳҜдёӢж»‘зҠ¶жҖҒпјҢе№¶дё”дёӢжӢүеӨҙжҳҜе®Ңе…Ёйҡҗи—Ҹзҡ„
                     *
                     * е°ұеұҸи”ҪдёӢжӢүдәӢд»¶
                     */
                    if (distance <= 0
                            && this.headerLayoutParams.topMargin <= this.hideHeaderHeight) {

                        PullToRefreshView.result = false;

                        return false;

                    }

                    // еҰӮжһңж»‘еҠЁзҡ„и·қзҰ» е°ҸдәҺ з®—жҳҜж»‘еҠЁзҡ„и·қзҰ»
                    if (distance < touchSlop) {

                        PullToRefreshView.result = false;

                        return false;

                    }

                    // еҪ“еүҚзҠ¶жҖҒ дёҚжҳҜ жӯЈеңЁеҲ·ж–°
                    if (this.currentStatus != PullToRefreshView.STATUS_REFRESHING) {

                        if (this.headerLayoutParams.topMargin > 0) {

                            this.currentStatus = PullToRefreshView.STATUS_RELEASE_TO_REFRESH;

                        } else {

                            this.currentStatus = PullToRefreshView.STATUS_PULL_TO_REFRESH;

                        }

                        // ж»‘еҠЁж—¶ еҰӮжһң headйғЁеҲҶ дёҚеҸҜи§Ғ
                        if (this.pull_to_refresh_linearLayout.getVisibility() == View.INVISIBLE) {

                            // и®ҫзҪ® еҸҜи§Ғ headйғЁеҲҶ
                            this.pull_to_refresh_linearLayout
                                    .setVisibility(View.VISIBLE);

                        }

                        // йҖҡиҝҮеҒҸз§»дёӢжӢүеӨҙзҡ„topMarginеҖјпјҢжқҘе®һзҺ°дёӢжӢүж•Ҳжһң
                        this.headerLayoutParams.topMargin = (distance / 2)
                                + this.hideHeaderHeight;

                        this.header.setLayoutParams(this.headerLayoutParams);

                    }
                }
                break;

            case MotionEvent.ACTION_UP:

            default:

                // еҰӮжһң еҪ“еүҚзҠ¶жҖҒ дёә йҮҠж”ҫз«ӢеҚіеҲ·ж–°зҠ¶жҖҒ
                if (this.currentStatus == PullToRefreshView.STATUS_RELEASE_TO_REFRESH) {

                    // жқҫжүӢж—¶еҰӮжһңжҳҜйҮҠж”ҫз«ӢеҚіеҲ·ж–°зҠ¶жҖҒпјҢе°ұеҺ»и°ғз”ЁжӯЈеңЁеҲ·ж–°зҡ„д»»еҠЎ
                    new RefreshingTask().execute();

                    // еҰӮжһң еҪ“еүҚзҠ¶жҖҒ дёәдёӢжӢүзҠ¶жҖҒ
                } else if (this.currentStatus == PullToRefreshView.STATUS_PULL_TO_REFRESH) {

                    // жқҫжүӢж—¶еҰӮжһңжҳҜдёӢжӢүзҠ¶жҖҒпјҢе°ұеҺ»и°ғз”Ёйҡҗи—ҸдёӢжӢүеӨҙзҡ„д»»еҠЎ
                    new HideHeaderTask().execute();

                }

                break;
            }

            /*
             * ж—¶еҲ»и®°еҫ—жӣҙж–°дёӢжӢүеӨҙдёӯзҡ„дҝЎжҒҜ
             *
             * еҰӮжһң еҪ“еүҚзҠ¶жҖҒ дёәдёӢжӢүзҠ¶жҖҒ or йҮҠж”ҫз«ӢеҚіеҲ·ж–°зҠ¶жҖҒ
             */
            if (this.currentStatus == PullToRefreshView.STATUS_PULL_TO_REFRESH
                    || this.currentStatus == PullToRefreshView.STATUS_RELEASE_TO_REFRESH) {

                // жӣҙж–°дёӢжӢүеӨҙдёӯзҡ„дҝЎжҒҜ
                this.updateHeaderView();

                // иҰҒи®©ListViewеӨұеҺ»з„ҰзӮ№пјҢеҗҰеҲҷиў«зӮ№еҮ»зҡ„йӮЈдёҖйЎ№дјҡдёҖзӣҙеӨ„дәҺйҖүдёӯзҠ¶жҖҒ
                this.listView.setPressed(false);

                this.listView.setFocusable(false);

                this.listView.setFocusableInTouchMode(false);

                this.lastStatus = this.currentStatus;

                // еҪ“еүҚжӯЈеӨ„дәҺдёӢжӢүжҲ–йҮҠж”ҫзҠ¶жҖҒпјҢйҖҡиҝҮиҝ”еӣһtrueеұҸи”ҪжҺүListViewзҡ„ж»ҡеҠЁдәӢд»¶

                return true;

            }

        }

        return false;
    }

    /**

     * @param id

    public void setOnRefreshListener(PullToRefreshListener listener, int id) {

    /**

    public void finishRefreshing() {

        this.currentStatus = PullToRefreshView.STATUS_REFRESH_FINISHED;

        this.preferences.edit()
                .putLong(UPDATED_AT + mId, System.currentTimeMillis()).commit();

        // и°ғз”Ёйҡҗи—ҸдёӢжӢүеӨҙзҡ„д»»еҠЎ

    /**
     * ж №жҚ®еҪ“еүҚListViewзҡ„ж»ҡеҠЁзҠ¶жҖҒжқҘи®ҫе®ҡ {@link #ableToPull}зҡ„еҖј
     *

     * иҝҷж ·еҸҜд»ҘеҲӨж–ӯеҮәеҪ“еүҚеә”иҜҘжҳҜж»ҡеҠЁListViewпјҢиҝҳжҳҜеә”иҜҘиҝӣиЎҢдёӢжӢүгҖӮ
     *
     * @param event
     */
    private void setIsAbleToPull(MotionEvent event) {

        View firstChild = this.listView.getChildAt(0);

        if (firstChild != null) {

            int firstVisiblePos = this.listView.getFirstVisiblePosition();

            if (firstVisiblePos == 0 && firstChild.getTop() == 0) {

                if (!this.ableToPull) {

                    this.yDown = event.getRawY();

                }

                /*
                 * еҰӮжһңйҰ–дёӘе…ғзҙ зҡ„дёҠиҫ№зјҳпјҢи·қзҰ»зҲ¶еёғеұҖеҖјдёә0

                 * е°ұиҜҙжҳҺListViewж»ҡеҠЁеҲ°дәҶжңҖйЎ¶йғЁпјҢжӯӨж—¶еә”иҜҘе…Ғи®ёдёӢжӢүеҲ·ж–°
                 */
                this.ableToPull = true;

            } else {

                if (this.headerLayoutParams.topMargin != this.hideHeaderHeight) {

                    this.headerLayoutParams.topMargin = this.hideHeaderHeight;

                    this.header.setLayoutParams(this.headerLayoutParams);

                }

                this.ableToPull = false;

            }

        } else {

            // еҰӮжһңListViewдёӯжІЎжңүе…ғзҙ пјҢд№ҹеә”иҜҘе…Ғи®ёдёӢжӢүеҲ·ж–°
            this.ableToPull = true;

        }

    }

    /**
     * жӣҙж–°дёӢжӢүеӨҙдёӯзҡ„дҝЎжҒҜ
     */
    private void updateHeaderView() {

        // еҰӮжһңдёҠж¬Ўзҡ„зҠ¶жҖҒ дёҚзӯүдәҺ иҝҷж¬Ўзҡ„зҠ¶жҖҒ
        if (this.lastStatus != this.currentStatus) {

            // "дёӢжӢүеҲ·ж–°зҠ¶жҖҒ"
            if (this.currentStatus == PullToRefreshView.STATUS_PULL_TO_REFRESH) {

                this.description.setText(getResources().getString(
                        R.string.pull_to_refresh));

                this.arrow.setVisibility(View.VISIBLE);

                this.progressBar.setVisibility(View.GONE);

                // ж №жҚ®еҪ“еүҚзҡ„зҠ¶жҖҒжқҘж—ӢиҪ¬з®ӯеӨҙ
                this.rotateArrow();

                // "йҮҠж”ҫз«ӢеҚіеҲ·ж–°зҠ¶жҖҒ"
            } else if (this.currentStatus == PullToRefreshView.STATUS_RELEASE_TO_REFRESH) {

                this.description.setText(getResources().getString(
                        R.string.release_to_refresh));

                this.arrow.setVisibility(View.VISIBLE);

                this.progressBar.setVisibility(View.GONE);

                // ж №жҚ®еҪ“еүҚзҡ„зҠ¶жҖҒжқҘж—ӢиҪ¬з®ӯеӨҙ
                this.rotateArrow();

                // "жӯЈеңЁеҲ·ж–°зҠ¶жҖҒ"
            } else if (this.currentStatus == PullToRefreshView.STATUS_REFRESHING) {

                this.description.setText(getResources().getString(
                        R.string.refreshing));

                this.progressBar.setVisibility(View.VISIBLE);

                this.arrow.clearAnimation();

                this.arrow.setVisibility(View.GONE);

            }

            // еҲ·ж–°дёӢжӢүеӨҙдёӯдёҠж¬Ўжӣҙж–°ж—¶й—ҙзҡ„ж–Үеӯ—жҸҸиҝ°
            this.refreshUpdatedAtValue();

        }

    }

    /**
     * ж №жҚ®еҪ“еүҚзҡ„зҠ¶жҖҒжқҘж—ӢиҪ¬з®ӯеӨҙ
     */
    private void rotateArrow() {

        float pivotX = arrow.getWidth() / 2f;

        float pivotY = arrow.getHeight() / 2f;

        float fromDegrees = 0f;

        float toDegrees = 0f;

        // "дёӢжӢүеҲ·ж–°зҠ¶жҖҒ"
        if (this.currentStatus == PullToRefreshView.STATUS_PULL_TO_REFRESH) {

            fromDegrees = 180f;

            toDegrees = 360f;

            // "йҮҠж”ҫз«ӢеҚіеҲ·ж–°зҠ¶жҖҒ"
        } else if (this.currentStatus == PullToRefreshView.STATUS_RELEASE_TO_REFRESH) {

            fromDegrees = 0f;

            toDegrees = 180f;

        }

        // ж—ӢиҪ¬еҠЁз”»
        RotateAnimation animation = new RotateAnimation(fromDegrees, toDegrees,
                pivotX, pivotY);

        animation.setDuration(100);

        animation.setFillAfter(true);

        arrow.startAnimation(animation);

    }

    /**
     * еҲ·ж–°дёӢжӢүеӨҙдёӯдёҠж¬Ўжӣҙж–°ж—¶й—ҙзҡ„ж–Үеӯ—жҸҸиҝ°
     */
    private void refreshUpdatedAtValue() {

        // д»Һ SharedPreferences еҸ–еҮә дёҠж¬Ўжӣҙж–°ж—¶й—ҙ
        this.lastUpdateTime = this.preferences.getLong(
                PullToRefreshView.UPDATED_AT + this.mId, -1);

        // еҸ–еҫ— д»ҘжҜ«з§’дёәеҚ•дҪҚзҡ„еҪ“еүҚж—¶й—ҙ
        long currentTime = System.currentTimeMillis();

        // и®Ўз®—ж—¶й—ҙе·®
        long timePassed = currentTime - this.lastUpdateTime;

        long timeIntoFormat;

        String updateAtValue;

        if (this.lastUpdateTime == -1) {

            updateAtValue = getResources().getString(R.string.not_updated_yet);

        } else if (timePassed < 0) {

            updateAtValue = getResources().getString(R.string.time_error);

        } else if (timePassed < PullToRefreshView.ONE_MINUTE) {

            updateAtValue = getResources().getString(R.string.updated_just_now);

        } else if (timePassed < PullToRefreshView.ONE_HOUR) {

            timeIntoFormat = timePassed / PullToRefreshView.ONE_MINUTE;

            String value = timeIntoFormat + "еҲҶй’ҹ";

            updateAtValue = String.format(
                    getResources().getString(R.string.updated_at), value);

        } else if (timePassed < PullToRefreshView.ONE_DAY) {

            timeIntoFormat = timePassed / PullToRefreshView.ONE_HOUR;

            String value = timeIntoFormat + "е°Ҹж—¶";

            updateAtValue = String.format(
                    getResources().getString(R.string.updated_at), value);

        } else if (timePassed < PullToRefreshView.ONE_MONTH) {

            timeIntoFormat = timePassed / PullToRefreshView.ONE_DAY;

            String value = timeIntoFormat + "еӨ©";

            updateAtValue = String.format(
                    getResources().getString(R.string.updated_at), value);

        } else if (timePassed < PullToRefreshView.ONE_YEAR) {

            timeIntoFormat = timePassed / PullToRefreshView.ONE_MONTH;

            String value = timeIntoFormat + "дёӘжңҲ";

            updateAtValue = String.format(
                    getResources().getString(R.string.updated_at), value);

        } else {

            timeIntoFormat = timePassed / PullToRefreshView.ONE_YEAR;

            String value = timeIntoFormat + "е№ҙ";

            updateAtValue = String.format(
                    getResources().getString(R.string.updated_at), value);

        }

        this.updateAt.setText(updateAtValue);

    }

    /**
     * жӯЈеңЁеҲ·ж–°зҡ„д»»еҠЎ
     *
     * еңЁжӯӨд»»еҠЎдёӯдјҡеҺ»еӣһи°ғжіЁеҶҢиҝӣжқҘзҡ„дёӢжӢүеҲ·ж–°зӣ‘еҗ¬еҷЁ
     *
     * @author CaMnter
     *
     */
    class RefreshingTask extends AsyncTask<Void, Integer, Void> {

        @Override
        protected Void doInBackground(Void... params) {

            int topMargin = PullToRefreshView.this.headerLayoutParams.topMargin;

            while (true) {

                topMargin = topMargin + PullToRefreshView.SCROLL_SPEED;

                if (topMargin <= 0) {

                    topMargin = 0;

                    break;

                }

                // жҜҸдёӘи°ғз”ЁжӯӨж–№жі•е°Ҷи§ҰеҸ‘onProgressUpdateеңЁUIзәҝзЁӢзҡ„жү§иЎҢ
                this.publishProgress(topMargin);

                PullToRefreshView.this.sleep(10);

            }

            PullToRefreshView.this.currentStatus = PullToRefreshView.STATUS_REFRESHING;

            this.publishProgress(0);

            PullToRefreshView.this.sleep(3000);

            PullToRefreshView.this.finishRefreshing();

            return null;

        }

        @Override
        protected void onPostExecute(Void result) {

            /**
             * еҰӮжһң doInBackground зҡ„onRefresh() дёӯжңүдәӢд»¶еӨ„зҗҶ йӮЈд№Ҳ жү§иЎҢдёҚеҲ° doInBackground йҮҢзҡ„
             * return null е°ұдёҖзӣҙеӨ„еңЁеҲ·ж–°зҠ¶жҖҒж•Ҳжһң жӯӨж—¶пјҢеҸҜд»ҘжҠҠ doInBackground йҮҢзҡ„ onRefreshж“ҚдҪңжҚўеҲ°
             * onPostExecuteйҮҢжқҘ
             **/
            if (PullToRefreshView.this.mListener != null) {

                // и°ғз”ЁеҲ·ж–°ж–№жі•
                PullToRefreshView.this.mListener.onRefresh();

            }

            super.onPostExecute(result);

        }

        @Override
        protected void onProgressUpdate(Integer... topMargin) {

            // жӣҙж–°дёӢжӢүеӨҙдёӯзҡ„дҝЎжҒҜ
            PullToRefreshView.this.updateHeaderView();

            PullToRefreshView.this.headerLayoutParams.topMargin = topMargin[0];

            PullToRefreshView.this.header
                    .setLayoutParams(PullToRefreshView.this.headerLayoutParams);

        }

    }

    /**
     * йҡҗи—ҸдёӢжӢүеӨҙзҡ„д»»еҠЎпјҢеҪ“жңӘиҝӣиЎҢдёӢжӢүеҲ·ж–°жҲ–дёӢжӢүеҲ·ж–°е®ҢжҲҗеҗҺ
     *
     * жӯӨд»»еҠЎе°ҶдјҡдҪҝдёӢжӢүеӨҙйҮҚж–°йҡҗи—Ҹ
     *
     * @author CaMnter
     *
     */
    class HideHeaderTask extends AsyncTask<Void, Integer, Integer> {

        @Override
        protected Integer doInBackground(Void... params) {

            int topMargin = PullToRefreshView.this.headerLayoutParams.topMargin;

            while (true) {

                topMargin = topMargin + PullToRefreshView.SCROLL_SPEED;

                if (topMargin <= PullToRefreshView.this.hideHeaderHeight) {

                    topMargin = PullToRefreshView.this.hideHeaderHeight;

                    break;

                }

                // жҜҸдёӘи°ғз”ЁжӯӨж–№жі•е°Ҷи§ҰеҸ‘onProgressUpdateеңЁUIзәҝзЁӢзҡ„жү§иЎҢ
                this.publishProgress(topMargin);

                PullToRefreshView.this.sleep(10);

            }

            return topMargin;

        }

        @Override
        protected void onProgressUpdate(Integer... topMargin) {

            PullToRefreshView.this.headerLayoutParams.topMargin = topMargin[0];

            PullToRefreshView.this.header
                    .setLayoutParams(PullToRefreshView.this.headerLayoutParams);

        }

        /**
         *
         * doInBackgroundеҗҺеңЁUIзәҝзЁӢдёҠиҝҗиЎҢ
         *
         * жҢҮе®ҡзҡ„з»“жһңжҳҜdoInBackgroundиҝ”еӣһзҡ„еҖјгҖӮ
         *
         * еҰӮжһңд»»еҠЎиў«еҸ–ж¶Ҳ,иҝҷдёӘж–№жі•дёҚдјҡиў«и°ғз”ЁгҖӮ
         *
         */
        @Override
        protected void onPostExecute(Integer topMargin) {

            PullToRefreshView.this.headerLayoutParams.topMargin = topMargin;

            PullToRefreshView.this.header
                    .setLayoutParams(PullToRefreshView.this.headerLayoutParams);

            // и®ҫзҪ®зҠ¶жҖҒдёә еҲ·ж–°з»“жқҹ
            PullToRefreshView.this.currentStatus = PullToRefreshView.STATUS_REFRESH_FINISHED;

        }

    }

    /**
     * дҪҝеҪ“еүҚзәҝзЁӢзқЎзң жҢҮе®ҡзҡ„жҜ«з§’ж•°
     *
     * @param time

    private void sleep(int time) {

        try {

            Thread.sleep(time);

            e.printStackTrace();

        }

    }

     *

    public interface PullToRefreshListener {

        /**
         * еҲ·ж–°ж—¶дјҡеҺ»еӣһи°ғжӯӨж–№жі•пјҢеңЁж–№жі•еҶ…зј–еҶҷе…·дҪ“зҡ„еҲ·ж–°йҖ»иҫ‘гҖӮ
         *
         * жіЁж„ҸжӯӨж–№жі•жҳҜеңЁеӯҗзәҝзЁӢдёӯи°ғз”Ёзҡ„пјҢ
         *
         * еҸҜд»ҘдёҚеҝ…еҸҰејҖзәҝзЁӢжқҘиҝӣиЎҢиҖ—ж—¶ж“ҚдҪңгҖӮ
         */
        void onRefresh();

    }

}

                zeroIcon);

        this.listView.setAdapter(this.exquisiteAdapter);

        pullToRefreshView.setOnRefreshListener(new PullToRefreshListener() {
            @Override
